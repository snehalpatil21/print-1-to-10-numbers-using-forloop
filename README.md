# print-1-to-10-numbers-using-forloop
using System;
using System.IO;
using System.Linq;
using System.Collections.Generic;

namespace CSharp_Shell
{

    public class Program 
    {
        public static void Main()
        {
			int i;
			for(i=1;i<=10;i++)
			{
				Console.WriteLine(i);
			}
        }
    }
}
